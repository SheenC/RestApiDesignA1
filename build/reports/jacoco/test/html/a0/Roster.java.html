package a0;

import java.util.*;

/**
 * Roster class for A0
 *
 * @author JasmineChu (AndrewID: xiyinc)
 */
public class Roster {
    /**
     * The course number in string format.
     */
    private String course;
    /**
     * A set of student andrewID that are registered in the course.
     */
    private Set<String> students;

    /*
     * Initialize the roster for a given course
     *
     * @param course the course number in string format
     */
    public Roster(String course) {
        this.course = course;
        this.students = new HashSet();
    }

    /*
     * Add a student andrewID into to roster
     *
     * @param andrewID the student that will be added
     * @return true if added successfully, false if the andrewID is null then we cannot add it
     */
    public boolean addStudent(String andrewID) {
        if (andrewID != null && andrewID.length() > 0) {
            students.add(andrewID);
            return true;
        }
        return false;
    }

    /*
     * Search if a student exists in the roster of the course
     *
     * @param andrewID the student that will be searched
     */
    public boolean searchStudent(String andrewID) {
        return students.contains(andrewID);
    }

    /*
     * Remove a student from a roster
     *
     * @param andrewID the student that will be removed from the roster
     * @return true if deleted successfully, false if the student is not in the roster then we cannot remove it
     */
    public boolean removeStudent(String andrewID) {
        // check if the student exists in the roster
        if (searchStudent(andrewID)) {
            students.remove(andrewID);
            return true;
        }
        return false;
    }

    public int getClassSize() {
        return this.students.size();
    }
}
